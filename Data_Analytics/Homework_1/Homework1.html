#Funzione per i primi n Fibonacci
fibo <- function(n){
  if(n == 0)
    return(-1)
  if(n == 1)
    return(0)
  
  ris <- c(0, 1)
  primo <- 0
  secondo <- 1
  count <- 2
  while(count < n){
    nuovo <- primo+secondo
    ris <- append(ris, nuovo)
    primo <- secondo
    secondo <- nuovo
    count <- count + 1
  }
  return(ris)
}

#Funzione che include anche somma e media
fibo2 <- function(n){
  if(n == 0)
    return(-1)
  if(n == 1)
    return(0)

  
  ris <- c(0, 1)
  primo <- 0
  secondo <- 1
  count <- 2
  while(count < n){
    nuovo <- primo+secondo
    ris <- append(ris, nuovo)
    primo <- secondo
    secondo <- nuovo
    count <- count + 1
  }
  media <- mean(ris)
  somma <- sum(ris)
  ris <- append(ris, c(somma, media))
  return(ris)
}
